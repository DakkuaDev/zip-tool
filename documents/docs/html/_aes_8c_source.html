174 static void Aes_Encode(const UInt32 *w, UInt32 *dest, const UInt32 *src)
175 {
176  UInt32 s[4];
177  UInt32 m[4];
178  UInt32 numRounds2 = w[0];
179  w += 4;
180  s[0] = src[0] ^ w[0];
181  s[1] = src[1] ^ w[1];
182  s[2] = src[2] ^ w[2];
183  s[3] = src[3] ^ w[3];
184  w += 4;
185  for (;;)

189  break;
190  HT16(s, m, 4);
191  w += 8;
192  }

195 }
