71 static SRes Xz_ReadIndex2(CXzStream *p, const Byte *buf, size_t size, ISzAlloc *alloc)
72 {
73  size_t i, numBlocks, crcStartPos, pos = 1;
74  UInt32 crc;
75 
76  if (size < 5 || buf[0] != 0)
77  return SZ_ERROR_ARCHIVE;
78 
79  size -= 4;
80  crc = CrcCalc(buf, size);
81  if (crc != GetUi32(buf + size))
82  return SZ_ERROR_ARCHIVE;
83 
84  {
85  UInt64 numBlocks64;
86  READ_VARINT_AND_CHECK(buf, pos, size, &numBlocks64);
87  numBlocks = (size_t)numBlocks64;
88  if (numBlocks != numBlocks64 || numBlocks * 2 > size)
89  return SZ_ERROR_ARCHIVE;
90  }
91 
92  crcStartPos = pos;
93  Xz_Free(p, alloc);
94  if (numBlocks != 0)
95  {
96  p->numBlocks = numBlocks;
97  p->numBlocksAllocated = numBlocks;
98  p->blocks = alloc->Alloc(alloc, sizeof(CXzBlockSizes) * numBlocks);
99  if (p->blocks == 0)
100  return SZ_ERROR_MEM;
101  for (i = 0; i < numBlocks; i++)
102  {
103  CXzBlockSizes *block = &p->blocks[i];
104  READ_VARINT_AND_CHECK(buf, pos, size, &block->totalSize);
105  READ_VARINT_AND_CHECK(buf, pos, size, &block->unpackSize);
106  if (block->totalSize == 0)
107  return SZ_ERROR_ARCHIVE;
108  }
109  }
110  while ((pos & 3) != 0)
111  if (buf[pos++] != 0)
112  return SZ_ERROR_ARCHIVE;
113  return (pos == size) ? SZ_OK : SZ_ERROR_ARCHIVE;
114 }

154  {

157  for (;;)

159  size_t i;
160  #define TEMP_BUF_SIZE (1 << 10)
161  Byte tempBuf[TEMP_BUF_SIZE];
162  if (*startOffset < XZ_STREAM_FOOTER_SIZE || total > (1 << 16))
163  return SZ_ERROR_NO_ARCHIVE;

165  total += (UInt32)i;

167  RINOK(SeekFromCur(stream, startOffset));
168  RINOK(LookInStream_Read2(stream, tempBuf, i, SZ_ERROR_NO_ARCHIVE));
169  for (; i != 0; i--)
170  if (tempBuf[i - 1] != 0)
171  break;
172  if (i != 0)
173  {
174  if ((i & 3) != 0)
175  return SZ_ERROR_NO_ARCHIVE;
176  *startOffset += i;
177  break;
178  }
179  }

181  return SZ_ERROR_NO_ARCHIVE;
182  *startOffset -= XZ_STREAM_FOOTER_SIZE;
183  RINOK(stream->Seek(stream, startOffset, SZ_SEEK_SET));
184  RINOK(LookInStream_Read2(stream, buf, XZ_STREAM_FOOTER_SIZE, SZ_ERROR_NO_ARCHIVE));

187  }

195  return SZ_ERROR_ARCHIVE;

198
