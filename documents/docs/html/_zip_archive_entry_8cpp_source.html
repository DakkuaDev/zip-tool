611  {

648  // if this entry is a directory, it should not contain any data

672  stream.seekp(this->GetCompressedSize(), std::ios::cur);

760 uint8_t ZipArchiveEntry::GetLastByteOfEncryptionHeader()
761 {
762  if (!!(this->GetGeneralPurposeBitFlag() & BitFlag::DataDescriptor))
763  {
764  // In the case that bit 3 of the general purpose bit flag is set to
765  // indicate the presence of a 'data descriptor' (signature
766  // 0x08074b50), the last byte of the decrypted header is sometimes
767  // compared with the high-order byte of the lastmodified time,
768  // rather than the high-order byte of the CRC, to verify the
769  // password.
770  //
771  // This is not documented in the PKWare Appnote.txt.
772  // This was discovered this by analysis of the Crypt.c source file in the
773  // InfoZip library
774  // http://www.info-zip.org/pub/infozip/
775 
776  // Also, winzip insists on this!
777  return uint8_t((_centralDirectoryFileHeader.LastModificationTime >> 8) & 0xff);
778  }
779  else
780  {
781  // When bit 3 is not set, the CRC value is required before
782  // encryption of the file data begins. In this case there is no way
